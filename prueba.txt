program superDemo;
var a, b, c: real;
var txt, res: string;
begin
  read("Numero inicial:, a");
  b := a ^ 2 - (-3 * (a + 4.5));
  res := subcadena(txt + "X", 2, longitud(txt));
  write("B=", b, ", Res=", res);
  if ((a < b) or (b >= 0.0)) and not buscar(res,"a") then
    write("Condicion A cumplida");
  else
    write("Condicion B");
  endif;
  while (c != 10) do
    c := c + 1;
    if (c == 5) then
      write("Llego a cinco");
    endif;
  endwhile;
end.
